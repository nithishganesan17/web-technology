body { font-family: Arial, sans-serif; background-color: #f0f0f0; padding: 20px; } .container { max-width: 400px; margin: auto; background-color: white; padding: 20px; border-radius: 5px; box-shadow: 0 0 10px rgba(0, 0, 0, 0.1); } h1 { text-align: center; color: #333; } label { margin-top: 10px; display: block; } input[type="text"], input[type="number"], input[type="submit"] { width: 100%; padding: 10px; margin: 10px 0; border: 1px solid #ccc; border-radius: 4px; } input[type="submit"] { background-color: #007BFF; color: white; border: none; cursor: pointer; } input[type="submit"]:hover { background-color: #0056b3; } .result-section { margin-top: 20px; padding: 10px; border: 1px solid #007BFF; border-radius: 4px; background-color: #e7f1ff; } document.getElementById('resultForm').addEventListener('submit', function (event) { event.preventDefault(); // Prevent the form from submitting // Get the values from the input fields const studentName = document.getElementById('studentName').value; const subject1 = parseFloat(document.getElementById('subject1').value); const subject2 = parseFloat(document.getElementById('subject2').value); const subject3 = parseFloat(document.getElementById('subject3').value); // Calculate average and grade const average = (subject1 + subject2 + subject3) / 3; let grade = ''; if (average >= 90) { grade = 'A'; } else if (average >= 80) { grade = 'B'; } else if (average >= 70) { grade = 'C'; } else if (average >= 60) { grade = 'D'; } else { grade = 'F'; } // Display the result document.getElementById('result').innerHTML = `

Results for ${studentName}

Average: ${average.toFixed(2)}

Grade: ${grade}

`; });
